Vulnerable Code — trading-data

# trading-data/app.py

@app.route('/internal/info')
def internal_info():
    """
    ⚠ Unauthenticated endpoint — exposes:
      - DB connection string (user + password)
      - Environment variables
      - Package versions
      - Server hostname
    """
    return jsonify({
        "db_url": os.environ.get("DATABASE_URL"),
        "env":    dict(os.environ),
        "version": {
            "python":  sys.version,
            "flask":   flask.__version__,
        },
        "host": socket.gethostname(),
    })

Safe Version

# Require authentication + return only safe info

@app.route('/internal/info')
@require_internal_token
def internal_info():
    return jsonify({
        "status":  "ok",
        "version": app_version,
        # Never expose env vars or connection strings
    })

Impact

  • Database credentials exposed → full data dump
  • API keys visible → impersonation of services
  • Exact package versions → targeted CVE exploitation
  • Internal hostnames → lateral movement mapping
