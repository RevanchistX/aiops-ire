# app.py

import os

@app.route('/cmd-injection', methods=['POST'])
def cmd_injection():
    target = request.form.get('target', '')
    ip     = request.remote_addr

    logger.warning(
        "CMD query=%s ip=%s", target, ip
    )

    # ⚠ Unsanitised input → shell execution
    output = os.popen(
        f"ping -c 1 {target}"
    ).read()

    return render_template(
        'cmd_injection.html', output=output
    )

Safe Version

import subprocess, re

SAFE_HOST = re.compile(
    r'^[a-zA-Z0-9.\-]{1,253}$'
)

def cmd_injection():
    target = request.form.get('target', '')
    if not SAFE_HOST.match(target):
        return "Invalid host", 400

    # subprocess with list args — no shell expansion
    result = subprocess.run(
        ["ping", "-c", "1", target],
        capture_output=True,
        text=True,
        timeout=5,
    )
    return render_template(..., output=result.stdout)
